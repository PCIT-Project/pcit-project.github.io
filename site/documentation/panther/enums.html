	NORTH,     // 0

	SOUTH = 4, // 4

type Color = enum(UI8) {
	RED   = 1 << 0,
	GREEN = 1 << 1,
	BLUE  = 1 << 2,

	func containsRed = (this) -> Bool {
		return (this & Color.RED) as Bool;
	}
}


func entry = () #entry -> UI8 {
	const purple: Color = Color.RED | Color.BLUE;
	return purple.containsRed() as UI8;
}
